rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {
    
    function isAuthenticated() {
      return request.auth != null;
    }
    
    function isUser(userId) {
      return request.auth.uid == userId;
    }
    
    // ===== USERS COLLECTION =====
    match /users/{userId} {
      allow read, write: if isUser(userId);
      allow update: if request.auth == null && request.resource.data.subscription != null;
      
      match /subscription/{document=**} {
        allow read, write: if isUser(userId);
      }
      
      match /settings/{document=**} {
        allow read, write: if isUser(userId);
      }
      
      match /preferences/{document=**} {
        allow read, write: if isUser(userId);
      }
    }

    // ===== CHAT SESSIONS COLLECTION =====
    match /chatSessions/{sessionId} {
      allow read: if isUser(resource.data.userId);
      allow create: if isAuthenticated() && request.resource.data.userId == request.auth.uid;
      allow update: if isUser(resource.data.userId);
      allow delete: if isUser(resource.data.userId);
      
      match /messages/{messageId} {
        allow read: if isUser(resource.data.userId);
        allow write: if isUser(resource.data.userId);
      }
    }

    // ===== FLASHCARD SETS COLLECTION =====
    match /studySets/{setId} {
      allow read: if isUser(resource.data.userId) || resource.data.isPublic == true;
      allow create: if isAuthenticated() && request.resource.data.userId == request.auth.uid;
      allow update, delete: if isUser(resource.data.userId);
      
      match /cards/{cardId} {
        allow read: if isUser(resource.data.userId);
        allow write: if isUser(resource.data.userId);
      }
      
      match /reviews/{reviewId} {
        allow read, write: if isUser(resource.data.userId);
      }
    }

    // ===== QUIZZES COLLECTION =====
    match /quizzes/{quizId} {
      allow read: if isUser(resource.data.userId) || resource.data.isPublic == true;
      allow create: if isAuthenticated() && request.resource.data.userId == request.auth.uid;
      allow update, delete: if isUser(resource.data.userId);
      
      match /attempts/{attemptId} {
        allow read, write: if isUser(resource.data.userId);
      }
    }

    // ===== STUDY GUIDES COLLECTION =====
    match /studyGuides/{guideId} {
      allow read: if isUser(resource.data.userId) || resource.data.isPublic == true;
      allow create: if isAuthenticated() && request.resource.data.userId == request.auth.uid;
      allow update, delete: if isUser(resource.data.userId);
      
      match /sections/{sectionId} {
        allow read: if isUser(resource.data.userId);
        allow write: if isUser(resource.data.userId);
      }
    }

    // ===== DOCUMENTS COLLECTION =====
    match /documents/{docId} {
      allow read: if isUser(resource.data.userId) || resource.data.isPublic == true;
      allow create: if isAuthenticated() && request.resource.data.userId == request.auth.uid;
      allow update, delete: if isUser(resource.data.userId);
    }

    // ===== PROGRESS TRACKING COLLECTION =====
    match /progress/{progressId} {
      allow read: if isUser(resource.data.userId);
      allow create: if isAuthenticated() && request.resource.data.userId == request.auth.uid;
      allow update, delete: if isUser(resource.data.userId);
    }

    // ===== STUDY ROOMS COLLECTION =====
    match /studyRooms/{roomId} {
      allow read: if resource.data.creatorId == request.auth.uid || resource.data.isPublic == true;
      allow create: if isAuthenticated();
      allow update, delete: if resource.data.creatorId == request.auth.uid;
      
      match /messages/{messageId} {
        allow read: if isAuthenticated();
        allow create, write: if isAuthenticated() && request.resource.data.userId == request.auth.uid;
      }
      
      match /participants/{participantId} {
        allow read: if isAuthenticated();
        allow write: if resource.parent().parent().data.creatorId == request.auth.uid;
      }
    }

    // ===== COMMUNITY POSTS COLLECTION =====
    match /communityPosts/{postId} {
      allow read: if resource.data.isPublic == true || isUser(resource.data.userId);
      allow create: if isAuthenticated() && request.resource.data.userId == request.auth.uid;
      allow update, delete: if isUser(resource.data.userId);
      
      match /comments/{commentId} {
        allow read: if resource.data.isPublic == true || isUser(resource.data.userId);
        allow create: if isAuthenticated() && request.resource.data.userId == request.auth.uid;
        allow update, delete: if isUser(resource.data.userId);
      }
      
      match /likes/{likeId} {
        allow read: if true;
        allow write: if isUser(likeId);
      }
    }

    // ===== NOTIFICATIONS COLLECTION =====
    match /notifications/{notifId} {
      allow read: if isUser(resource.data.userId);
      allow write: if request.auth == null && resource.data.userId != null;
      allow delete: if isUser(resource.data.userId);
    }

    // ===== ACTIVITY LOGS =====
    match /activityLogs/{logId} {
      allow read: if isUser(resource.data.userId);
      allow create: if isAuthenticated() && request.resource.data.userId == request.auth.uid;
      allow delete: if isUser(resource.data.userId);
    }

    // ===== PAYMENT HISTORY =====
    match /paymentHistory/{paymentId} {
      allow read: if isUser(resource.data.userId);
      allow create: if request.auth == null;
      allow delete: if false;
    }

    // ===== DEFAULT DENY ALL =====
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
